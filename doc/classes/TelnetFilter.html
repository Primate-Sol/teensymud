Class TelnetFilter
In: lib/protocol/telnetfilter.rb
Parent: Filter

The TelnetFilter class implements the Telnet protocol.

This implements most of basic Telnet as per RFC 854/855/1143 and options in RFC 857/858/1073/1091

 67:     init_subneg
 68:     return "" if str.nil? || str.empty?

 71:     @sc ? @sc.concat(str) : @sc = StringScanner.new(str)
 72:     while b = @sc.get_byte
 73: 
 74:       # OOB sync data
 75:       if @pstack.urgent_on || b[0] == DM
 76:         @pstack.log.debug("(#{@pstack.conn.object_id}) Sync mode on")
 77:         @pstack.urgent_on = false
 78:         @synch = true
 79:         break
 80:       end
 81: 
 82:       case mode?
 83:       when :normal
 84:         case b[0]
 85:         when CR
 86:           next if @synch
 87:           set_mode(:cr) if !@pstack.binary_on
 88:         when IAC
 89:           set_mode(:cmd)
 90:         when NUL  # ignore NULs in stream when in normal mode
 91:           next if @synch
 92:           if @pstack.binary_on
 93:             buf << b
 94:             echo(b)
 95:           else
 96:             @pstack.log.error("(#{@pstack.conn.object_id}) unexpected NUL found in stream")
 97:           end
 98:         when BS, DEL
 99:           next if @synch
100:           buf << b
101:           echo(BS.chr)
102:         else
103:           next if @synch
104:           # Only let 7-bit values through in normal mode
105:           if (b[0] & 0x80 == 0) && !@pstack.binary_on
106:             buf << b
107:             echo(b)
108:           else
109:             @pstack.log.error("(#{@pstack.conn.object_id}) unexpected 8-bit byte found in stream '#{b[0]}'")
110:           end
111:         end
112:       when :cr
113:         # handle CRLF and CRNUL by swallowing what follows CR and
114:         # insertion of LF
115:         if !@synch
116:           case b[0]
117:           when LF, NUL
118:             buf << LF.chr
119:             echo(CR.chr + LF.chr)
120:           else  # Handle screwed up muds
121:             buf << CR.chr + b
122:             echo(CR.chr + b)
123:           end
124:         end
125:         set_mode(:normal)
126:       when :cmd
127:         case b[0]
128:         when IAC
129:           # IAC escapes IAC
130:           buf << IAC.chr
131:           set_mode(:normal)
132:         when AYT
133:           @pstack.log.debug("(#{@pstack.conn.object_id}) AYT sent - Msg returned")

135:           set_mode(:normal)
136:         when AO

139:           @pstack.conn.sock.send(IAC.chr + DM.chr, 0)
140:           @pstack.conn.sockio.write_urgent(DM.chr)
141:           set_mode(:normal)
142:         when IP
143:           @pstack.conn.sockio.read_flush
144:           @pstack.conn.sockio.write_flush
145:           @pstack.log.debug("(#{@pstack.conn.object_id}) IP sent")

148:           @pstack.log.debug("(#{@pstack.conn.object_id}) GA, NOP or BRK sent")
149:           set_mode(:normal)

156:           @pstack.log.debug("(#{@pstack.conn.object_id}) EC sent")
