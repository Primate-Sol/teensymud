Class TelnetFilter
In: lib/protocol/telnetfilter.rb
Parent: Filter

The TelnetFilter class implements the Telnet protocol.

This implements most of basic Telnet as per RFC 854/855/1143 and options in RFC 857/858/1073/1091

 77: 
 78:       # OOB sync data
 79:       if @pstack.urgent_on || b[0] == DM

 83:         break
 84:       end

 87:       when :normal
 88:         case b[0]
 89:         when CR
 90:           next if @synch
 91:           set_mode(:cr) if !@pstack.binary_on
 92:         when LF  # LF or LF/CR may be issued by broken mud servers and clients
 93:           next if @synch
 94:           set_mode(:lf) if !@pstack.binary_on
 95:           buf << LF.chr
 96:           echo(CR.chr + LF.chr)

 98:           set_mode(:cmd)
 99:         when NUL  # ignore NULs in stream when in normal mode

106:           end
107:         when BS, DEL
108:           next if @synch
109:           # Leaves BS, DEL in input stream for higher filter to deal with.
110:           buf << b
111:           echo(BS.chr)
112:         else
113:           next if @synch
114:           # Only let 7-bit values through in normal mode
115:           if (b[0] & 0x80 == 0) && !@pstack.binary_on
116:             buf << b
117:             echo(b)
118:           else
119:             log.error("(#{@pstack.conn.object_id}) unexpected 8-bit byte found in stream '#{b[0]}'")
120:           end
121:         end
122:       when :cr
123:         # handle CRLF and CRNUL by insertion of LF
124:         case b[0]
125:         when LF, NUL
126:           buf << LF.chr
127:           echo(CR.chr + LF.chr)
128:         else # eat lone CR
129:           buf << b
130:           echo(b)
131:         end
132:         set_mode(:normal)
133:       when :lf
134:         # handle LF, LFCR
