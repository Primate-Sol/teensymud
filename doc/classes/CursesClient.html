105:   def run
106:     shutdown = false
107:     connection = Reactor.new(@opts.port, $connopts, @opts.address)
108:     raise "Unable to start TeensyClient" unless connection.start(self)
109:     conmsg "Connected to #{@opts.address}:#{@opts.port}.  Use F10 to QUIT"
110:     until shutdown
111:       connection.poll(0.1)
112:       Curses.refresh
113:       c = Curses.getch
114:       case c
115:       when 32..127
116:         publish(c.chr)
117:       when Curses::KEY_ENTER
118:         publish("\r\n")
119:       when 10
120:         publish("\n")
121:       when 4294967295 # Error Timeout. This is -1 in Bignum format
122:       when Curses::KEY_F10
123:         conmsg "Quitting..."
124:         shutdown = true
125:       else
126:         conmsg "Unknown key hit code - #{c.inspect}"
127:       end
128:     end # until
129:     connection.stop
130:   rescue SystemExit, Interrupt
131:     conmsg "\nConnection closed exiting"
132:   rescue Exception
