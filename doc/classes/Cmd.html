cmd_kill(*args)

This kills a player anywhere - it’s link death (30% chance)

[Source]

    # File cmd/cmd_kill.rb, line 16
16:   def cmd_kill(*args)
17:     case args[0]
18:     when nil
19:       sendto("Who do you want to kill?")
20:     else
21:       d = $world.find_player_by_name(args[0])
22:       if !d
23:         sendto("Can't find them.")
24:         return
25:       end
26:       if rand < 0.3
27:         sendto("You kill #{d.name}.")
28:         $world.players_connected(@oid).each {|p|
29:           $world.add_event(@oid,p.oid,:show,"#{@name} kills #{d.name}.")
30:         }
31:         d.disconnect
32:         # $world.delete(d)  Dont delete player, it's annoying
33:       else

36:           $world.add_event(@oid,p.oid,:show,"#{@name} attacks and misses #{d.name}.")
