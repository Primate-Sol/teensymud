cmd_fart(args)

adds or deletes a farts trigger to an object Syntax: 

  @fart add #<id> <progname> <eventtype>
  @fart del #<id> <eventtype>
  @fart show #<id>

(ex. @fart add 1 myprog arrive)

[Source]

    # File cmd/teensy/cmd_fart.rb, line 21
21:   def cmd_fart(args)
22:     case args
23:     when nil, ""
24:       sendto("What??")
25:     when /del\s+#(\d+)\s+(\w+)/
26:       o = $engine.world.db.get($1.to_i)
27:       case o
28:       when nil, 0
29:         sendto("No object.")
30:       else
31:         t = o.get_trigger($2)
32:         if t
33:           o.delete_trigger($2)
34:           sendto("Object #" + $1 + " fart trigger deleted.")
35:         else
36:           sendto("Trigger #{$2} not found on object.")
37:         end
38:       end
39:     when /add\s+#(\d+)\s+(\w+)\s+(\w+)/
40:       o = $engine.world.db.get($1.to_i)
41:       case o
42:       when nil, 0
43:         sendto("No object.")
44:       else
45:         t = Farts::FartTrigger.new($2,$3)
46:         o.add_trigger(t)
47:         sendto("Object #" + $1 + " fart trigger added.")
48:       end
49:     when /show\s+#(\d+)/
50:       o = $engine.world.db.get($1.to_i)
51:       case o
52:       when nil, 0
53:         sendto("No object.")
54:       else
55:         sendto("====================TRIGGERS=====================")
56:         sendto("Program                  Event          Compiled?")
57:         o.get_triggers.each do |t|
58:           sendto(sprintf("%-25s %-15s %s", t.fname, t.event.id2name, t.prog ? "yes" : "no"))
59:         end
60:       end
61:     else
62:       sendto("What??")
63:     end
64:   end
