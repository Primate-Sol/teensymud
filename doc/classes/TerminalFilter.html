 44:     buf = ""
 45: 
 46:     str.each_byte do |b|
 47:       case mode?
 48:       when :normal
 49:         case b
 50:         when BS, DEL
 51:           @pstack.log.debug("(#{@pstack.conn.object_id}) BS, DEL found")
 52:           if buf.size > 1
 53:             buf.slice!(-1)
 54:           elsif @pstack.conn.inbuffer.size > 0
 55:             @pstack.conn.inbuffer.slice!(-1)
 56:           end
 57:         when TAB
 58:           @pstack.log.debug("(#{@pstack.conn.object_id}) TAB found")
 59:           buf << "[TAB]"
 60:         when BEL
 61:           buf << "[BELL]"
 62:         when ESC
 63:           @pstack.log.debug("(#{@pstack.conn.object_id}) ESC found")
 64:           set_mode :escape
 65:         else
 66:           buf << b.chr
 67:         end
 68:       when :escape
 69:         case b
 70:         when ?[
 71:           @csi = ""
 72:           @pstack.log.debug("(#{@pstack.conn.object_id}) CSI sequence found")
 73:           set_mode :csi
 74:         when ?O
 75:           set_mode :ss3
 76:         when ?D
 77:           buf << "[SCROLL DOWN]"
 78:           set_mode :normal
 79:         when ?U
 80:           buf << "[SCROLL UP]"
 81:           set_mode :normal
 82:         else
 83:           set_mode :normal
 84:         end
 85:       when :csi
 86:         case b
 87:         when ?A
 88:           i = @csi.to_i
 89:           i = 1 if i == 0
 90:           @csi = ""
 91:           buf << "[UP #{i}]"
 92:           set_mode :normal
 93:         when ?B
 94:           i = @csi.to_i
 95:           i = 1 if i == 0
 96:           @csi = ""
 97:           buf << "[DOWN #{i}]"
 98:           set_mode :normal
 99:         when ?C

102:           @csi = ""
103:           buf << "[RIGHT #{i}]"
104:           set_mode :normal
105:         when ?D
106:           i = @csi.to_i
107:           i = 1 if i == 0
108:           @csi = ""
109:           buf << "[LEFT #{i}]"
110:           set_mode :normal
111:         when ?H, ?f
112:           a = @csi.split(";")
113:           @csi = ""
