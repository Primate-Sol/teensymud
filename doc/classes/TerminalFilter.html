     # File lib/protocol/terminalfilter.rb, line 338
338:   def filter_out(str)
339:     VTKeys.each do |key,val|
340:       while str =~ key do
341:         s = val.dup
342:         p1 = $1.dup if $1
343:         p2 = $2.dup if $2
344:         if p1 && p2
345:           s.sub!(/\$;\$/, "#{p1};#{p2}")
346:         elsif p1
347:           s.sub!(/\$/, p1)
348:         end
349:         str.sub!(key,s)
350:       end
351:     end

353:   end
