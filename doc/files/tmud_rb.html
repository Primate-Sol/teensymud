173:   # parse options
174:   begin
175:     # The myopts specified on the command line will be collected in *myopts*.
176:     # We set default values here.
177:     myopts = OpenStruct.new

182:     myopts.trace = false
183: 
184:     opts = OptionParser.new do |opts|
185:       opts.banner = BANNER
186:       opts.separator ""
187:       opts.separator "Usage: ruby #{$0} [options]"
188:       opts.separator ""
189:       opts.on("-p", "--port PORT", Integer,
190:         "Select the port the mud will run on",
191:         "  (defaults to 4000)") {|myopts.port|}
192:       opts.on("-d", "--database DBNAME", String,
193:         "Select the name of the database the mud will use",
194:         "  (defaults to \'db/world.yaml\')") {|myopts.dbname|}
195:       opts.on("-h", "--home LOCATIONID", Integer,
196:         "Select the object id where new players will start",
197:         "  (defaults to 1)") {|myopts.home|}
198:       opts.on("-t", "--[no-]trace", "Trace execution") {|myopts.trace|}
199:       opts.on("-v", "--[no-]verbose", "Run verbosely") {|myopts.verbose|}
200:       opts.on_tail("-h", "--help", "Show this message") do
201:         $stdout.puts opts.help
202:         exit
203:       end
204:       opts.on_tail("--version", "Show version") do
205:         $stdout.puts "TeensyMud #{Version}"
206:         exit
207:       end
208:     end
209: 
210:     opts.parse!(ARGV)
211: 
212:     return myopts
213:   rescue OptionParser::ParseError
214:     $stderr.puts "ERROR - #{$!}"
215:     $stderr.puts "For help..."
216:     $stderr.puts " ruby #{$0} --help"
217:     exit
218:   end
219: end
